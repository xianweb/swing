# 新建项目

  1. 新建一个java project项目

  2. 新建一个项目包com.seecen.fruit

  3. 在com.seecen.fruit中新建2个子包app和util

  4. 在app子包中新建window子包

  5. 在util包下新建 Cons.java配置文件和GUITool.java工具文件

  6. 在window包下新建AbstractEnterFrame.java抽象入口界面类和EnterFrame.java入口界面实现类

  7. AbstractEnterFrame文件负责创建入口界面、添加相关组件、按钮事件

  8. AbstractEnterFrame文件将单击进入系统的方法抽象出来，方便覆盖。

  9. EnterFrame.java 改写showMainFrame方法

  10. 单击进入系统按钮后，打开水果超市主界面

  11. 在window子包下创建AbstractMainFrame.java抽象主界面类

  12. AbstractMainFrame.java抽象类主要实现 标题组件、表格组件、文本框组件、按钮等添加，将刷新查询事件、按钮添加事件、按钮修改事件、按钮删除事件定义为抽象方法。

  13. MainFrame.java 对AbstractMainFrame.java抽象类的方法进行覆盖，分别覆盖 查询事件、添加事件、修改事件、删除事件

# AbstractEnterFrame入口界面抽象类

package com.seecen.fruit.app.window;

import java.awt.BorderLayout;
import java.awt.Color;
import java.awt.HeadlessException;
import java.awt.event.ActionEvent;
import java.awt.event.ActionListener;

import javax.swing.ImageIcon;
import javax.swing.JButton;
import javax.swing.JFrame;
import javax.swing.JLabel;
import javax.swing.JPanel;

import com.seecen.fruit.util.GUITool;

public abstract class AbstractEnterFrame extends JFrame{

	private JButton btn = new JButton("进入系统");
	
	public AbstractEnterFrame() throws HeadlessException {
		this.init();
		this.addComponent();
		this.addListener();
	}
	
	private void init() {
		GUITool.createFrame(this, "水果超市欢迎您!", "mini");
	}

	private void addComponent() {
		JLabel label = new JLabel(new ImageIcon("./images/banner.jpg"));
		this.add( label, BorderLayout.NORTH );
		
		JPanel panel = new JPanel();
		panel.setBackground(Color.white);
		panel.add(btn);
		
		this.add(panel);
	}
	
	private void addListener() {
		btn.addActionListener( new ActionListener() {

			@Override
			public void actionPerformed(ActionEvent e) {
				showMainFrame();
			}
			
		});
	}
	
	public abstract void showMainFrame();
	

}

package com.seecen.fruit.app.window;

public class EnterFrame extends AbstractEnterFrame {

# AbstractMainFrame入口界面抽象类

package com.seecen.fruit.app.window;

import com.seecen.fruit.app.pojo.Fruit;

import java.awt.BorderLayout;
import java.awt.Dimension;

import java.awt.Font;
import java.awt.GridLayout;
import java.awt.HeadlessException;
import java.awt.event.ActionEvent;
import java.awt.event.ActionListener;

import javax.swing.JButton;
import javax.swing.JComboBox;
import javax.swing.JFrame;
import javax.swing.JLabel;
import javax.swing.JPanel;
import javax.swing.JScrollPane;
import javax.swing.JTable;
import javax.swing.JTextField;

import com.seecen.fruit.util.GUITool;

public abstract class AbstractMainFrame extends JFrame {

	private JLabel title = new JLabel("水果列表", JLabel.CENTER);
	
	protected JTable table = new JTable();
	private JScrollPane scroll = new JScrollPane(table);
	
	private JLabel numberLabel = new JLabel("水果编号");
	private JLabel nameLabel = new JLabel("水果名称");
	private JLabel priceLabel = new JLabel("水果单价");
	private JLabel unitLabel = new JLabel("计价单位");
	private JLabel actionLabel = new JLabel("操作");
	    
    protected JTextField addNumberText = new JTextField();
    protected JTextField addNameText = new JTextField();
    protected JTextField addPriceText = new JTextField();
//    protected JTextField addUnitText = new JTextField();
    protected JComboBox<String> addUnitText = new JComboBox<String>();
    
    private JButton addButton = new JButton("添加水果");
    
    private JTextField updateNumberText = new JTextField();
    private JTextField updateNameText = new JTextField();
    private JTextField updatePriceText = new JTextField();
//    private JTextField updateUnitText = new JTextField();
    protected JComboBox<String> updateUnitText = new JComboBox<String>();
    
    private JButton updateButton = new JButton("修改水果");
    
    private JTextField deleteNumberText = new JTextField();

    private JButton deleteButton = new JButton("删除水果");
    private JLabel spaceLabel1 = new JLabel("");
    private JLabel spaceLabel2 = new JLabel("");
    private JLabel spaceLabel3 = new JLabel("");
	private JPanel footer = new JPanel();

	public AbstractMainFrame() throws HeadlessException {
		this.init();
		this.addComponent();
		this.addBtnListener();
	}

	private void init() {
		GUITool.createFrame(this, "超市货物管理", "default");
	}

	private void addComponent() {
				
		title.setPreferredSize(new Dimension(0, 60));
		title.setFont(new Font(Font.SANS_SERIF, Font.BOLD, 18));
		
		footer.setLayout(new GridLayout(4,5,10,10));
		
		footer.add(numberLabel);
		footer.add(nameLabel);
		footer.add(priceLabel);
		footer.add(unitLabel);
		footer.add(actionLabel);
		
		footer.add(addNumberText);
		footer.add(addNameText);
		footer.add(addPriceText);
		

		footer.add(addUnitText);
		footer.add(addButton);
		
		footer.add(updateNumberText);
		footer.add(updateNameText);
		footer.add(updatePriceText);
		footer.add(updateUnitText);
		footer.add(updateButton);
		
		footer.add(deleteNumberText);
		footer.add(spaceLabel1);
		footer.add(spaceLabel2);
		footer.add(spaceLabel3);
		footer.add(deleteButton);

		this.add(title, BorderLayout.NORTH);
		this.add(scroll, BorderLayout.CENTER);
		this.add(footer, BorderLayout.SOUTH);
		
	}

    protected void addBtnListener() {
        addButton.addActionListener(new ActionListener() {
            public void actionPerformed(ActionEvent e) {
                addFruitItem();
            }
        });
        updateButton.addActionListener(new ActionListener() {
            public void actionPerformed(ActionEvent e) {
                updateFruitItem();
            }
        });
        deleteButton.addActionListener(new ActionListener() {
            public void actionPerformed(ActionEvent e) {
                deleteFruitItem();
            }
        });
    }
	
    private Fruit getFruit(JTextField numberText, JTextField nameText, JTextField priceText, JComboBox<String> unitText) {
        Fruit fruit = new Fruit();
        fruit.setNumber(Integer.parseInt(numberText.getText().trim()));
        fruit.setName(nameText.getText().trim());
        fruit.setPrice(Double.parseDouble(priceText.getText()));
        fruit.setUnit((String) unitText.getSelectedItem() );
        return fruit;
    }
    
    protected Fruit getAddFruit() {
        return getFruit(addNumberText, addNameText, addPriceText, addUnitText);
    }
    
    protected Fruit getUpdateFruit() {
        return getFruit(updateNumberText, updateNameText, updatePriceText, updateUnitText);
    }
    
    protected Integer getDeleteNumber() {
        return Integer.parseInt(deleteNumberText.getText().trim());
    }
	
	public abstract void queryFruitItem();
    public abstract void updateFruitItem();
    public abstract void deleteFruitItem();
    public abstract void addFruitItem();

}
