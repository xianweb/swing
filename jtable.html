# JTable 表格组件

表格是 Swing 新增加的组件，主要功能是把数据以二维表格的形式显示出来，并且允许用户对表格中的数据进行编辑。

# 创建基本实例

# 源代码

package hello;

import java.awt.Font;
import java.awt.event.ActionEvent;
import java.awt.event.ActionListener;

import javax.swing.JButton;
import javax.swing.JComboBox;
import javax.swing.JFrame;
import javax.swing.JLabel;
import javax.swing.JOptionPane;
import javax.swing.JPanel;
import javax.swing.JPasswordField;
import javax.swing.JScrollPane;
import javax.swing.JTable;
import javax.swing.JTextField;

public class MainApp {

	public static void main(String[] args) {

		JFrame frame = new JFrame("水果超市欢迎您!");
		frame.setBounds(0, 0, 400, 200);

		JPanel panel = new JPanel();
		
		//创建3行4列的表格
		JTable table=new JTable(3,4);
		JScrollPane pane=new JScrollPane(table);
		
		panel.add(pane);

		frame.add(panel);

		frame.setVisible(true);
	}

}

# 动态创建表格数据

# 源代码

package hello;

import java.awt.BorderLayout;
import java.awt.Font;
import java.awt.event.ActionEvent;
import java.awt.event.ActionListener;

import javax.swing.JButton;
import javax.swing.JComboBox;
import javax.swing.JFrame;
import javax.swing.JLabel;
import javax.swing.JOptionPane;
import javax.swing.JPanel;
import javax.swing.JPasswordField;
import javax.swing.JScrollPane;
import javax.swing.JTable;
import javax.swing.JTextField;
import javax.swing.table.DefaultTableModel;

public class MainApp {

	public static void main(String[] args) {

		JFrame frame = new JFrame("水果超市欢迎您!");
		frame.setBounds(0, 0, 460, 400);

		JPanel panel = new JPanel(new BorderLayout());

		// 创建3行4列的表格
		JTable table = new JTable();
		JScrollPane pane = new JScrollPane(table);

		JButton btn = new JButton("加入表格数据");
		btn.addActionListener(new ActionListener() {

			@Override
			public void actionPerformed(ActionEvent e) {
				Object[][] tbody = new Object[5][8];
				for (int i = 0; i < 5; i++) {
					tbody[i][0] = "1000" + i;
					for (int j = 1; j < 8; j++) {
						tbody[i][j] = 0;
					}
				}
				String[] thead = { "学号", "软件工程", "Java", "网络", "数据结构", "数据库", "总成绩", "平均成绩" };
				table.setModel(new DefaultTableModel(tbody, thead));
			}

		});
		panel.add(pane, BorderLayout.NORTH);

		panel.add(btn, BorderLayout.SOUTH);

		frame.add(panel);

		frame.setVisible(true);
	}

}
