# JPanel 面板组件

JPanel 是一种中间层容器，它能容纳组件并将组件组合在一起，但它本身必须添加到其他容器中使用。

# 创建示例

# 源代码

package hello;

import java.awt.Color;

import javax.swing.JFrame;
import javax.swing.JLabel;
import javax.swing.JPanel;

public class MainApp {

	public static void main(String[] args) {

		JFrame frame = new JFrame("用户登录");
		frame.setBounds(300, 100, 400, 300);

		JPanel panel = new JPanel();
		panel.setBackground(Color.red);

		JLabel label = new JLabel("这是放在JPanel上的标签");
		
		panel.add(label);
		frame.add(panel);
		
		frame.setVisible(true);
	}

}

# JPanel 常用构造方法

# JPanel()：使用默认的布局管理器创建新面板，默认的布局管理器为 FlowLayout。

JPanel panel = new JPanel();

# JPanel( LayoutManagerLayout layout )：创建指定布局管理器的 JPanel 对象。

JPanel panel = new JPanel( new BorderLayout() );

# add()：将指定的组件追加到此容器的尾部

JPanel panel = new JPanel();
JLabel label = new JLabel("这是放在JPanel上的标签");
panel.add(label);

# setBackground(Color c)：设置背景颜色

JPanel panel = new JPanel();
panel.setBackground(Color.red);

# setLayout(LayoutManager mgr)：设置容器的布局管理器

JPanel panel = new JPanel();
panel.setLayout( new BorderLayout() );
