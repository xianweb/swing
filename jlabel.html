# JLabel 标签组件

标签是一种可以包含文本和图片的非交互组件，其文本可以是单行文本，也可以是 HTML 文本。

# 创建实例

# 源代码

package hello;

import javax.swing.JFrame;
import javax.swing.JLabel;

public class MainApp {

	public static void main(String[] args) {

		JFrame frame = new JFrame("用户登录");
		frame.setBounds(0, 0, 400, 300);

		JLabel label = new JLabel("我是标签!");

		frame.add(label);

		frame.setVisible(true);
	}

}

# 实际效果

# JLabel 常用构造方法

# JLabel()：创建无图像并且标题为空字符串的 JLabel。

# JLabel(Icon image)：创建具有指定图像的 JLabel。

JFrame frame = new JFrame("水果超市欢迎您!");
JLabel label = new JLabel(new ImageIcon("./images/banner.jpg"));
frame.add(label);

# JLabel(String text)：创建具有指定文本的 JLabel。

JFrame frame = new JFrame("用户登录");
JLabel label = new JLabel("我是标签!");
frame.add(label);

# JLabel(String textjcon image,int Alignment)：创建具有指定文本、图像和水平对齐方式的 JLabel

horizontalAlignment 有三个

  • JLabel.LEFT 左对齐
  • JLabel.RIGHT 右对齐
  • JLabel.CENTER 居中对齐

# 源代码

JFrame frame = new JFrame("用户登录");
JLabel label = new JLabel("我是标签!",JLabel.CENTER);
frame.add(label);

# 实际效果

# setFont 设置字体

JFrame frame = new JFrame("用户登录");
JLabel label = new JLabel("我是标签!",JLabel.CENTER);
label.setFont(new Font("楷体", Font.BOLD, 18));
frame.add(label);

# setPreferredSize 设置尺寸

JFrame frame = new JFrame("用户登录");
JLabel label = new JLabel("我是标签!",JLabel.CENTER);
label.setPreferredSize(new Dimension(60, 60));
frame.add(label);
